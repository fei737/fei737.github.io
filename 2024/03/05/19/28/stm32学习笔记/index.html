GPIO 开启时钟

RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA,ENABLE);//打开GPIO的时钟GPIO_InitTypeDef 
GPIO_InitStructure;
GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
GPIO_InitStructure.GPIO_Pin = GPIO_Pin_All;
GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
GPIO_Init(GPIOA,&GPIO_InitStructure);

#define OLED_CHN_CHAR_WIDTH 3 //UTF-8 编码格式给 3，GB2312 编码格式给 2

OLED_Printf(0,0,OLED_6X8,"Num1:%d,Num2:%d",123,-456);
OLED_Printf(0,10,OLED_6X8,"%x",0xA5A5);
